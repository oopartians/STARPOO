function lookAngle(ship,angle){
	angle%=360;
	if(Math.abs(angle-ship.angle) > Math.abs(ship.angle-(angle-360))){
		angle -= 360;
	}
	var v = (angle-ship.angle)/dt;

	return v < 360;
}

function lookPos(ship,pos,angle){
	if(angle == null){
		angle = 0;
	}
	p = polarFrom(ship,pos);
	var a = p.angle-angle;
	a %= 360;
	if(a > 180){
		a -= 360;
	}
	else if(a < -180){
		a += 360;
	}
	var v = a/dt;
	ship.setAngleSpeed(v);
	return v < 15;
}
